Reads a file in the current working directory or a String as a plain text toml file. The returned object is a normal Map with String keys or a List of primitives or Map.

Example:

    def datas = readToml file: 'dir/input.toml'
    assert datas['title'] == 'TOML Example'
    assert datas.title == 'TOML Example'

    def datas = readToml text: ''' title = "TOML Example" '''
    assert datas['title'] == 'TOML Example'
    assert datas.title == 'TOML Example'

    def datas = readToml text: 'ports = [ 8000, 8001, 8002 ]'
    assert datas.ports[0] == 8000
    assert datas.ports[-1] == 8002
Example:
    def datas = readToml text: '''# This is a TOML document

title = "TOML Example"

[owner]
name = "Tom Preston-Werner"
dob = 1979-05-27T07:32:00-08:00

[database]
enabled = true
ports = [ 8000, 8001, 8002 ]
data = [ ["delta", "phi"], [3.14] ]
temp_targets = { cpu = 79.5, case = 72.0 }

[servers]

[servers.alpha]
ip = "10.0.0.1"
role = "frontend"

[servers.beta]
ip = "10.0.0.2"
role = "backend"
'''
    assert datas.owner.name == "Tom Preston-Werner" // string
    println datas.owner.dob // Date

    assert datas.database.enabled == true  // boolean
    assert datas.database.ports[0] == 8000 // int number
    assert datas.database.ports[1] == 8001
    assert datas.database.ports[2] == 8002
    assert datas.database.temp_targets.cpu == 79.5  // float number
    assert datas.database.temp_targets.case == 72.0

    assert datas.servers.alpha.ip == "10.0.0.1"
    assert datas.servers.alpha.role == "frontend"
    assert datas.servers.beta.ip == "10.0.0.2"
    assert datas.servers.beta.role == "backend"
